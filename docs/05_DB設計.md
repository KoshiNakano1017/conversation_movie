## 2. テーブル定義

### 2.1 users
```sql
CREATE TABLE users (
  id          UUID PRIMARY KEY DEFAULT auth.uid(),
  email       TEXT NOT NULL UNIQUE,
  display_name TEXT,
  avatar_url  TEXT,
  youtube_refresh_token TEXT,  -- 暗号化して保存
  created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Supabase AuthのユーザーとRLS連携
ALTER TABLE users ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can view own profile"
  ON users FOR SELECT USING (auth.uid() = id);
CREATE POLICY "Users can update own profile"
  ON users FOR UPDATE USING (auth.uid() = id);
```

### 2.2 jobs
```sql
CREATE TYPE job_status AS ENUM (
  'pending',
  'uploading',
  'transcribing',
  'diarizing',
  'analyzing',
  'generating_subtitles',
  'generating_video',
  'generating_shorts',
  'uploading_youtube',
  'completed',
  'failed',
  'cancelled'
);

CREATE TABLE jobs (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  title           TEXT NOT NULL,
  description     TEXT,
  status          job_status NOT NULL DEFAULT 'pending',
  error_message   TEXT,
  progress        INTEGER DEFAULT 0 CHECK (progress >= 0 AND progress <= 100),
  
  -- タイムスタンプ
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  completed_at    TIMESTAMPTZ,
  
  -- Celeryタスク追跡
  celery_task_id  TEXT,
  
  -- 設定
  language        TEXT DEFAULT 'ja',
  speaker_count   INTEGER,  -- NULLは自動検出
  
  INDEX idx_jobs_user_id (user_id),
  INDEX idx_jobs_status (status),
  INDEX idx_jobs_created_at (created_at DESC)
);

ALTER TABLE jobs ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can manage own jobs"
  ON jobs FOR ALL USING (auth.uid() = user_id);
```

### 2.3 audio_files
```sql
CREATE TABLE audio_files (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id          UUID NOT NULL REFERENCES jobs(id) ON DELETE CASCADE,
  user_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  
  -- ファイル情報
  original_filename TEXT NOT NULL,
  storage_path    TEXT NOT NULL,  -- Supabase Storage パス
  storage_bucket  TEXT NOT NULL DEFAULT 'audio',
  file_size_bytes BIGINT NOT NULL,
  duration_seconds FLOAT,
  format          TEXT NOT NULL,  -- mp3, wav, mp4, etc.
  
  -- 処理状態
  is_processed    BOOLEAN DEFAULT FALSE,
  
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

ALTER TABLE audio_files ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can manage own audio files"
  ON audio_files FOR ALL USING (auth.uid() = user_id);
```

### 2.4 transcriptions
```sql
CREATE TABLE transcriptions (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id          UUID NOT NULL REFERENCES jobs(id) ON DELETE CASCADE,
  
  -- 文字起こし結果
  full_text       TEXT NOT NULL,
  language        TEXT NOT NULL DEFAULT 'ja',
  
  -- Whisper設定
  model_name      TEXT NOT NULL DEFAULT 'medium',
  word_error_rate FLOAT,  -- 評価値（あれば）
  
  -- タイムスタンプ付きセグメント
  segments        JSONB NOT NULL DEFAULT '[]',
  -- segments形式:
  -- [{"id": 0, "start": 0.0, "end": 5.2, "text": "こんにちは", "words": [...]}]
  
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

ALTER TABLE transcriptions ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can view own transcriptions"
  ON transcriptions FOR SELECT
  USING (EXISTS (
    SELECT 1 FROM jobs WHERE jobs.id = transcriptions.job_id
    AND jobs.user_id = auth.uid()
  ));
```

### 2.5 speaker_segments
```sql
CREATE TABLE speaker_segments (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id          UUID NOT NULL REFERENCES jobs(id) ON DELETE CASCADE,
  
  -- 話者情報
  speaker_label   TEXT NOT NULL,   -- "SPEAKER_00", "SPEAKER_01", etc.
  speaker_name    TEXT,            -- ユーザーが設定した名前
  
  -- セグメント
  start_time      FLOAT NOT NULL,
  end_time        FLOAT NOT NULL,
  text            TEXT NOT NULL,
  
  -- 信頼スコア
  confidence      FLOAT,
  
  -- 順序
  segment_index   INTEGER NOT NULL,
  
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  
  INDEX idx_speaker_segments_job_id (job_id),
  INDEX idx_speaker_segments_start_time (job_id, start_time)
);

ALTER TABLE speaker_segments ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can view own speaker segments"
  ON speaker_segments FOR SELECT
  USING (EXISTS (
    SELECT 1 FROM jobs WHERE jobs.id = speaker_segments.job_id
    AND jobs.user_id = auth.uid()
  ));
```

### 2.6 analyses
```sql
CREATE TABLE analyses (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id          UUID NOT NULL REFERENCES jobs(id) ON DELETE CASCADE UNIQUE,
  
  -- 要約
  summary_short   TEXT NOT NULL,
  summary_medium  TEXT NOT NULL,
  summary_detailed JSONB NOT NULL DEFAULT '{}',
  -- {
  --   "overview": "...",
  --   "key_decisions": ["..."],
  --   "action_items": [{"who": "...", "what": "...", "when": "..."}],
  --   "next_steps": ["..."]
  -- }
  
  -- テーマ・キーワード
  themes          TEXT[] NOT NULL DEFAULT '{}',
  keywords        TEXT[] NOT NULL DEFAULT '{}',
  
  -- 名言
  quotes          JSONB NOT NULL DEFAULT '[]',
  -- [{"speaker": "...", "text": "...", "timestamp": "00:05:23"}]
  
  -- 感情分析
  sentiment_timeline JSONB NOT NULL DEFAULT '[]',
  -- [{"timestamp": "00:01:00", "sentiment": "positive", "score": 0.8}]
  overall_sentiment TEXT,
  
  -- YouTube メタデータ候補
  suggested_title TEXT,
  suggested_description TEXT,
  suggested_tags  TEXT[] DEFAULT '{}',
  
  -- Geminiモデル情報
  model_name      TEXT NOT NULL DEFAULT 'gemini-1.5-flash',
  tokens_used     INTEGER,
  
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

ALTER TABLE analyses ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can view own analyses"
  ON analyses FOR SELECT
  USING (EXISTS (
    SELECT 1 FROM jobs WHERE jobs.id = analyses.job_id
    AND jobs.user_id = auth.uid()
  ));
```

### 2.12 job_logs
```sql
CREATE TABLE job_logs (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id          UUID NOT NULL REFERENCES jobs(id) ON DELETE CASCADE,
  
  level           TEXT NOT NULL DEFAULT 'info',  -- 'info', 'warning', 'error'
  step            TEXT NOT NULL,  -- 処理ステップ名
  message         TEXT NOT NULL,
  details         JSONB DEFAULT '{}',
  
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  
  INDEX idx_job_logs_job_id (job_id),
  INDEX idx_job_logs_created_at (job_id, created_at)
);
```

## 3. ER図（テキスト形式）

```
users
  ├── jobs (1:N)
  │     ├── audio_files (1:1)
  │     ├── transcriptions (1:1)
  │     ├── speaker_segments (1:N)
  │     ├── analyses (1:1)
  │     ├── subtitles (1:N)
  │     ├── avatar_scripts (1:N)
  │     ├── videos (1:N)
  │     │     └── youtube_publications (1:1)
  │     └── job_logs (1:N)
  └── youtube_publications (1:N) [user_id参照]

avatar_characters (マスターデータ)
  └── avatar_scripts (N:N)
```

## 6. Supabase Storageバケット設計

| バケット名 | 用途 | アクセス制御 | 最大ファイルサイズ |
|---|---|---|---|
| audio | 音声ファイル | 認証ユーザーのみ | 500MB |
| videos | 生成動画 | 認証ユーザーのみ | 2GB |
| thumbnails | サムネイル | 公開 | 5MB |
| avatars | アバター画像 | 公開 | 10MB |

```sql
-- Storageのパス規則
-- audio/{user_id}/{job_id}/original.{ext}
-- videos/{user_id}/{job_id}/full.mp4
-- videos/{user_id}/{job_id}/shorts.mp4
-- thumbnails/{user_id}/{job_id}/thumbnail.jpg
```
